```python
import subprocess
import sys
import os
import numpy as np
import argparse

def build_correlation_graph(data_dir: str, threshold: float = 0.3) -> int:
    """
    Build filtered correlation graph and save corr_adj.npy + data.edgelist.
    Uses threshold to reduce edge count for 500-stock universe.
    Returns number of edges written.
    """
    import pandas as pd
    label_path = os.path.join(data_dir, 'label.csv')
    df = pd.read_csv(label_path, index_col=0)
    df.fillna(0, inplace=True)

    # Zero-variance check (same as existing script)
    epsilon = 1e-10
    std_devs = np.std(df.values, axis=0)
    zero_var = std_devs < epsilon
    df.iloc[:, zero_var] = epsilon

    corr_matrix = np.corrcoef(df.values, rowvar=False)
    np.save(os.path.join(data_dir, 'corr_adj.npy'), corr_matrix)

    # Build filtered edge list
    edge_list = []
    for i in range(corr_matrix.shape[0]):
        for j in range(i + 1, corr_matrix.shape[1]):
            weight = corr_matrix[i, j]
            if abs(weight) > threshold:
                edge_list.append((i, j, weight))

    edgelist_path = os.path.join(data_dir, 'data.edgelist')
    with open(edgelist_path, 'w') as f:
        for edge in edge_list:
            f.write(f'{edge[0]} {edge[1]} {edge[2]}\n')
    print(f'Edge list saved: {len(edge_list)} edges (threshold |corr|>{threshold})')
    return len(edge_list)

def run_struc2vec(data_dir: str, embed_size: int = 128, workers: int = 4) -> None:
    """
    Run Struc2Vec on the data.edgelist and save 128_corr_struc2vec_adjgat.npy.
    Delegates to the existing preprocessing script which has the full Struc2Vec pipeline.
    Falls back to direct ge import if the script is not found.
    """
    preprocess_script = os.path.join(
        os.path.dirname(__file__),
        '../../data_processing_script/stockformer_input_data_processing/Stockformer_data_preprocessing_script.py'
    )
    preprocess_script = os.path.normpath(preprocess_script)

    try:
        import ge  # noqa: F401
    except ImportError:
        raise ImportError(
            "GraphEmbedding library not installed. Run:\n"
            "pip install git+https://github.com/shenweichen/GraphEmbedding.git"
        )

    # The existing script reads label.csv (already present), recomputes corr, writes embedding
    # We already wrote the filtered edgelist above; the script will overwrite with its own
    # (unfiltered) edgelist, but then train Struc2Vec on it. We patch by calling ge directly.
    import networkx as nx
    from ge import Struc2Vec

    edgelist_path = os.path.join(data_dir, 'data.edgelist')
    G = nx.read_edgelist(
        edgelist_path, create_using=nx.DiGraph(),
        nodetype=None, data=[('weight', float)]
    )
    model = Struc2Vec(G, num_walks=10, walk_length=80, workers=workers, verbose=40)
    model.train(embed_size=embed_size)
    embeddings = model.get_embeddings()

    # Sort by node id to get consistent [N, 128] array
    n_nodes = len(embeddings)
    embedding_array = np.zeros((n_nodes, embed_size), dtype=np.float32)
    for node_id, vec in embeddings.items():
        embedding_array[int(node_id)] = vec

    out_path = os.path.join(data_dir, '128_corr_struc2vec_adjgat.npy')
    np.save(out_path, embedding_array)
    print(f'Embedding saved: shape {embedding_array.shape} → {out_path}')

def main():
    parser = argparse.ArgumentParser(description='Build Struc2Vec graph embedding')
    parser.add_argument('--data_dir', default='./data/Stock_SP500_2018-01-01_2024-01-01')
    parser.add_argument('--threshold', type=float, default=0.3)
    parser.add_argument('--embed_size', type=int, default=128)
    parser.add_argument('--workers', type=int, default=4)
    args = parser.parse_args()

    build_correlation_graph(args.data_dir, threshold=args.threshold)
    run_struc2vec(args.data_dir, embed_size=args.embed_size, workers=args.workers)

if __name__ == '__main__':
    main()
```

```python
#!/usr/bin/env python
"""
End-to-end S&P500 data pipeline orchestrator.

Runs all five steps in order:
  1. download_ohlcv.py     — fetch OHLCV from yfinance
  2. feature_engineering.py — compute TA features + label.csv
  3. normalize_split.py    — cross-sectional normalization + date split
  4. serialize_arrays.py   — save flow.npz + trend_indicator.npz
  5. graph_embedding.py    — build Struc2Vec embedding

Usage:
  python scripts/build_pipeline.py --data_dir ./data/Stock_SP500_2018-01-01_2024-01-01

Each step is skippable if its output already exists (idempotent reruns).
"""
import argparse
import subprocess
import sys
import os

PIPELINE_DIR = os.path.join(
    os.path.dirname(__file__), '..', 'data_processing_script', 'sp500_pipeline'
)

STEPS = [
    ('download_ohlcv.py',      lambda d: os.path.exists(os.path.join(d, 'tickers.txt'))),
    ('feature_engineering.py', lambda d: os.path.exists(os.path.join(d, 'label.csv'))),
    ('normalize_split.py',     lambda d: os.path.exists(os.path.join(d, 'split_indices.json'))),
    ('serialize_arrays.py',    lambda d: os.path.exists(os.path.join(d, 'flow.npz'))),
    ('graph_embedding.py',     lambda d: os.path.exists(os.path.join(d, '128_corr_struc2vec_adjgat.npy'))),
]


def run_step(script_name: str, data_dir: str, skip_if_exists_fn, force: bool = False) -> None:
    script_path = os.path.normpath(os.path.join(PIPELINE_DIR, script_name))
    if not force and skip_if_exists_fn(data_dir):
        print(f'[SKIP] {script_name} — output already exists')
        return
    print(f'\n[RUN] {script_name}')
    result = subprocess.run(
        [sys.executable, script_path, '--data_dir', data_dir],
        check=True
    )
    print(f'[DONE] {script_name} — exit code {result.returncode}')


def main():
    parser = argparse.ArgumentParser(description='S&P500 data pipeline orchestrator')
    parser.add_argument('--data_dir', default='./data/Stock_SP500_2018-01-01_2024-01-01')
    parser.add_argument('--start', default='2018-01-01')
    parser.add_argument('--end', default='2024-01-01')
    parser.add_argument('--force', action='store_true', help='Re-run all steps even if output exists')
    args = parser.parse_args()

    os.makedirs(args.data_dir, exist_ok=True)

    for script_name, skip_fn in STEPS:
        run_step(script_name, args.data_dir, skip_fn, force=args.force)

    print('\n[COMPLETE] All pipeline steps finished.')
    print(f'Output directory: {args.data_dir}')


if __name__ == '__main__':
    main()
```
